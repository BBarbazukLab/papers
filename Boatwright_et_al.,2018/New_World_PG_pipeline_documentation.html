14. Flag Significant Bayesian Poisson-Gamma Output¶

Flag those putative orthologous pairs that exhibit mapping bias. Flag significant if the 
    credible interval [similar to 95% Highest posterior density interval (HPDI)] does not overlap 0.5 
    for each prior (q4,q5,q6). 

Then, flag significant for all three if q4 = q5 = q6 = 1

Scripts:¶

execute_bayesian_flag_sig_results_parents.bash

Dependencies:¶

bayesian_flag_sig_results_parents_pdf.sas

Program Versions:¶

Input:¶

Output:¶

empirical_bayesian_parents_output/PG_emp_bayesian_results_{SP}_parents_flag_sig.csv

$\bar\theta$ plots:¶

  • $\bar\theta$ represents mapping bias for Reads onto corresponding references
  • Contain histogram of $\bar\theta$ values
  • Contain kernel density lines -- which represent a non-parametric way to estimate the probability density function of the $\bar\theta$ values
  • 0 == Line == T. dubius
  • 1 == Tester == T. pratensis or T. porrifolius

def filter_csv(file_name):
    """Given flagged bayesian output in CSV format, isolate unbiased loci. 
    This is only meant for the parents, as the parent data can biasedly map to its own reference or exhibit no bias."""
    import pandas as pd
    import numpy as np
    df = pd.read_csv(file_name, index_col = "commonID")
    cols = list(df.columns)
    sig_col = cols[-1]
    reads=sig_col.split("_")[2]
    unbiased_loci = []
    print("\tDF pre-filter: " + str(len(df.index)))
    if reads == "tdu":
        # identify loci not biased toward Alt
        not_biased = df[(df[sig_col] == 0) & (df["q5_mean_theta"] <= 0.75)]
        print("\t\tNot biased: " + str(len(not_biased.index)))
        df = pd.concat([df[(df["q5_mean_theta"] <= 0.5) & (df[sig_col] == 1)],not_biased])
    elif ((reads == "tpo") or (reads == "tpr")):
        # identify loci not biased toward TDU
        not_biased = df[(df[sig_col] == 0) & (df["q5_mean_theta"] >= 0.25)]
        print("\t\tNot biased: " + str(len(not_biased.index)))
        df = pd.concat([df[(df["q5_mean_theta"] >= 0.5) & (df[sig_col] == 1)], not_biased])
    print("\tDF post-filter: " + str(len(df.index)))
    return df

def csv_to_df(file_name):
    """Given flagged bayesian output in CSV format, convert to standard dataframe. 
    This is only meant for the hybrid, as the hybrid data may biasedly express one ortholog over another."""
    import pandas as pd
    import numpy as np
    df = pd.read_csv(file_name, index_col = "commonID")
    cols = list(df.columns)
    cols[-1]="sig_all_three"
    df.columns = cols
    print("Length of dataframe pre-filter: " + str(len(df["sig_all_three"])))
    return df

def hist_unfiltered_filtered(df, unbiased_loci):
    """Plot a histogram of both the unfiltered and filtered q5 mean theta to visualize filtering"""
    import numpy as np
    import matplotlib.pyplot as plt
    plt.figure()
    unbiased_df = df[df.index.isin(unbiased_loci)]
    print("\tLoci exhibiting HSE pre-filter: " + str(np.sum(df["sig_all_three"])))
    n, bins, patches = plt.hist(df["q5_mean_theta"], bins=50, histtype='step', facecolor='r', label='pre-filter')
    print("Total loci post-filter: " + str(len(unbiased_df["sig_all_three"])))
    print("\tLoci exhibiting HSE post-filter: " + str(np.sum(unbiased_df["sig_all_three"])))
    n, bins, patches = plt.hist(unbiased_df["q5_mean_theta"], bins=50, histtype='step', facecolor='b', label='post-filter')

    plt.xlabel('Mean Theta')
    plt.ylabel('Frequency')
    plt.legend()
    plt.title('Distribution of Mean Theta for Hybrid')
    plt.rcParams['figure.figsize'] = (8.0, 8.0)
    plt.show()
    return unbiased_df
